``` r
library(ggplot2)

# 创建一个包含变量的数据框
data <- data.frame(
  x = c(1, 2, 3, 4),
  y = c(10, 20, 30, 40),
  group = c("A", "A", "B", "B")
)

# 定义一个变量，用于替换图形属性
color_var <- "group"

# 使用变量替换创建散点图
ggplot(data, aes(x = x, y = y, color = !!as.name(color_var))) +
  geom_point()
```
